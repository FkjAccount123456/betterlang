func fac(num) {
    # Print("fac", num, "\n");
    if num == 0 {
        return 1;
    }
    return num * fac(num - 1);
}

func prime(num) {
    if num < 2 { return 0; }
    var i = 2;
    while i < num {
        # Print(num, " ", i, "\n");
        if num % i == 0 {
            # Print(i, "\n");
            return 0;
        }
        i = i + 1;
    }
    return 1;
}
